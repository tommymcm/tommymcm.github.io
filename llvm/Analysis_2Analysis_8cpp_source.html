97LLVMBool LLVMVerifyModule(LLVMModuleRef M, LLVMVerifierFailureAction Action,
98 char **OutMessages) {
99 raw_ostream *DebugOS = Action != LLVMReturnStatusAction ? &errs() : nullptr;
100 std::string Messages;
101 raw_string_ostream MsgsOS(Messages);
102
103 LLVMBool Result = verifyModule(*unwrap(M), OutMessages ? &MsgsOS : DebugOS);
104
105 // Duplicate the output to stderr.
106 if (DebugOS && OutMessages)
107 *DebugOS << MsgsOS.str();
108
109 if (Action == LLVMAbortProcessAction && Result)
110 report_fatal_error("Broken module found, compilation aborted!");
111
112 if (OutMessages)
113 *OutMessages = strdup(MsgsOS.str().c_str());
114
115 return Result;
116}

118LLVMBool LLVMVerifyFunction(LLVMValueRef Fn, LLVMVerifierFailureAction Action) {
119 LLVMBool Result = verifyFunction(
120 *unwrap<Function>(Fn), Action != LLVMReturnStatusAction ? &errs()
121 : nullptr);
122
123 if (Action == LLVMAbortProcessAction && Result)
124 report_fatal_error("Broken function found, compilation aborted!");
125
126 return Result;
127}
