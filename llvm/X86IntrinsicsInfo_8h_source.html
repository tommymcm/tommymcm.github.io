41struct IntrinsicData {
42
43 uint16_t Id;
44 IntrinsicType Type;
45 uint16_t Opc0;
46 uint16_t Opc1;
47
48 bool operator<(const IntrinsicData &RHS) const {
49 return Id < RHS.Id;
50 }
51 bool operator==(const IntrinsicData &RHS) const {
52 return RHS.Id == Id;
53 }
54 friend bool operator<(const IntrinsicData &LHS, unsigned Id) {
55 return LHS.Id < Id;
56 }
57};
