43void BPFSubtarget::initSubtargetFeatures(StringRef CPU, StringRef FS) {
44 if (CPU == "probe")
45 CPU = sys::detail::getHostCPUNameForBPF();
46 if (CPU == "generic" || CPU == "v1")
47 return;
48 if (CPU == "v2") {
49 HasJmpExt = true;
50 return;
51 }
52 if (CPU == "v3") {
53 HasJmpExt = true;
54 HasJmp32 = true;
55 return;
56 }
57}
