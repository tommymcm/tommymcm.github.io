36MCOperand BPFMCInstLower::LowerSymbolOperand(const MachineOperand &MO,
37 MCSymbol *Sym) const {
38
39 const MCExpr *Expr = MCSymbolRefExpr::create(Sym, Ctx);
40
41 if (!MO.isJTI() && MO.getOffset())
42 llvm_unreachable("unknown symbol op");
43
44 return MCOperand::createExpr(Expr);
45}

47void BPFMCInstLower::Lower(const MachineInstr *MI, MCInst &OutMI) const {
48 OutMI.setOpcode(MI->getOpcode());
49
50 for (unsigned i = 0, e = MI->getNumOperands(); i != e; ++i) {
51 const MachineOperand &MO = MI->getOperand(i);
52
53 MCOperand MCOp;
54 switch (MO.getType()) {
55 default:
56 MI->print(errs());
57 llvm_unreachable("unknown operand type");
58 case MachineOperand::MO_Register:
59 // Ignore all implicit register operands.
60 if (MO.isImplicit())
61 continue;
62 MCOp = MCOperand::createReg(MO.getReg());
63 break;
64 case MachineOperand::MO_Immediate:
65 MCOp = MCOperand::createImm(MO.getImm());
66 break;
67 case MachineOperand::MO_MachineBasicBlock:
68 MCOp = MCOperand::createExpr(
69 MCSymbolRefExpr::create(MO.getMBB()->getSymbol(), Ctx));
70 break;
71 case MachineOperand::MO_RegisterMask:
72 continue;
73 case MachineOperand::MO_ExternalSymbol:
74 MCOp = LowerSymbolOperand(MO, GetExternalSymbolSymbol(MO));
75 break;
76 case MachineOperand::MO_GlobalAddress:
77 MCOp = LowerSymbolOperand(MO, GetGlobalAddressSymbol(MO));
78 break;
79 }
80
81 OutMI.addOperand(MCOp);
82 }
83}
