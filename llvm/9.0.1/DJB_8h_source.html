20/// The Bernstein hash function used by the DWARF accelerator tables.
21inline uint32_t djbHash(StringRef Buffer, uint32_t H = 5381) {
22 for (unsigned char C : Buffer.bytes())
23 H = (H << 5) + H + C;
24 return H;
25}
