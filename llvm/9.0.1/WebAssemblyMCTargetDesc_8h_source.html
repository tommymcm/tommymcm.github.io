358inline unsigned GetDefaultP2Align(unsigned Opc) {
359 auto Align = GetDefaultP2AlignAny(Opc);
360 if (Align == -1U) {
361 llvm_unreachable("Only loads and stores have p2align values");
362 }
363 return Align;
364}
