23bool llvm::shouldOptimizeForSize(Function *F, ProfileSummaryInfo *PSI,
24 BlockFrequencyInfo *BFI) {
25 assert(F);
26 if (!PSI || !BFI || !PSI->hasProfileSummary())
27 return false;
28 return ProfileGuidedSizeOpt && PSI->isFunctionColdInCallGraph(F, *BFI);
29}
