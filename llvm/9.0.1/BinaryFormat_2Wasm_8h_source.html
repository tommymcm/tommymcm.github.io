333 I64 = WASM_TYPE_I64,

340struct WasmSignature {
341 SmallVector<ValType, 1> Returns;
342 SmallVector<ValType, 4> Params;
343 // Support empty and tombstone instances, needed by DenseMap.
344 enum { Plain, Empty, Tombstone } State = Plain;
345
346 WasmSignature(SmallVector<ValType, 1> &&InReturns,
347 SmallVector<ValType, 4> &&InParams)
348 : Returns(InReturns), Params(InParams) {}
349 WasmSignature() = default;
350};
351
352// Useful comparison operators
353inline bool operator==(const WasmSignature &LHS, const WasmSignature &RHS) {
354 return LHS.State == RHS.State && LHS.Returns == RHS.Returns &&
355 LHS.Params == RHS.Params;
356}
357
358inline bool operator!=(const WasmSignature &LHS, const WasmSignature &RHS) {
359 return !(LHS == RHS);
360}
361
362inline bool operator==(const WasmGlobalType &LHS, const WasmGlobalType &RHS) {
363 return LHS.Type == RHS.Type && LHS.Mutable == RHS.Mutable;
364}
365
366inline bool operator!=(const WasmGlobalType &LHS, const WasmGlobalType &RHS) {
367 return !(LHS == RHS);
368}
