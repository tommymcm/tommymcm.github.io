22class AArch64MCExpr : public MCTargetExpr {
23public:
24 enum VariantKind {
25 // Symbol locations specifying (roughly speaking) what calculation should be
26 // performed to construct the final address for the relocated
27 // symbol. E.g. direct, via the GOT, ...
28 VK_ABS = 0x001,
29 VK_SABS = 0x002,
30 VK_GOT = 0x003,
31 VK_DTPREL = 0x004,
32 VK_GOTTPREL = 0x005,
33 VK_TPREL = 0x006,
34 VK_TLSDESC = 0x007,
35 VK_SECREL = 0x008,
36 VK_SymLocBits = 0x00f,
37
38 // Variants specifying which part of the final address calculation is
39 // used. E.g. the low 12 bits for an ADD/LDR, the middle 16 bits for a
40 // MOVZ/MOVK.
41 VK_PAGE = 0x010,
42 VK_PAGEOFF = 0x020,
43 VK_HI12 = 0x030,
44 VK_G0 = 0x040,
45 VK_G1 = 0x050,
46 VK_G2 = 0x060,
47 VK_G3 = 0x070,
48 VK_AddressFragBits = 0x0f0,
49
50 // Whether the final relocation is a checked one (where a linker should
51 // perform a range-check on the final address) or not. Note that this field
52 // is unfortunately sometimes omitted from the assembly syntax. E.g. :lo12:
53 // on its own is a non-checked relocation. We side with ELF on being
54 // explicit about this!
55 VK_NC = 0x100,
56
57 // Convenience definitions for referring to specific textual representations
58 // of relocation specifiers. Note that this means the "_NC" is sometimes
59 // omitted in line with assembly syntax here (VK_LO12 rather than VK_LO12_NC
60 // since a user would write ":lo12:").
61 VK_CALL = VK_ABS,
62 VK_ABS_PAGE = VK_ABS | VK_PAGE,
63 VK_ABS_PAGE_NC = VK_ABS | VK_PAGE | VK_NC,
64 VK_ABS_G3 = VK_ABS | VK_G3,
65 VK_ABS_G2 = VK_ABS | VK_G2,
66 VK_ABS_G2_S = VK_SABS | VK_G2,
67 VK_ABS_G2_NC = VK_ABS | VK_G2 | VK_NC,
68 VK_ABS_G1 = VK_ABS | VK_G1,
69 VK_ABS_G1_S = VK_SABS | VK_G1,
70 VK_ABS_G1_NC = VK_ABS | VK_G1 | VK_NC,
71 VK_ABS_G0 = VK_ABS | VK_G0,
72 VK_ABS_G0_S = VK_SABS | VK_G0,
73 VK_ABS_G0_NC = VK_ABS | VK_G0 | VK_NC,
74 VK_LO12 = VK_ABS | VK_PAGEOFF | VK_NC,
75 VK_GOT_LO12 = VK_GOT | VK_PAGEOFF | VK_NC,
76 VK_GOT_PAGE = VK_GOT | VK_PAGE,
77 VK_DTPREL_G2 = VK_DTPREL | VK_G2,
78 VK_DTPREL_G1 = VK_DTPREL | VK_G1,
79 VK_DTPREL_G1_NC = VK_DTPREL | VK_G1 | VK_NC,
80 VK_DTPREL_G0 = VK_DTPREL | VK_G0,
81 VK_DTPREL_G0_NC = VK_DTPREL | VK_G0 | VK_NC,
82 VK_DTPREL_HI12 = VK_DTPREL | VK_HI12,
83 VK_DTPREL_LO12 = VK_DTPREL | VK_PAGEOFF,
84 VK_DTPREL_LO12_NC = VK_DTPREL | VK_PAGEOFF | VK_NC,
85 VK_GOTTPREL_PAGE = VK_GOTTPREL | VK_PAGE,
86 VK_GOTTPREL_LO12_NC = VK_GOTTPREL | VK_PAGEOFF | VK_NC,
87 VK_GOTTPREL_G1 = VK_GOTTPREL | VK_G1,
88 VK_GOTTPREL_G0_NC = VK_GOTTPREL | VK_G0 | VK_NC,
89 VK_TPREL_G2 = VK_TPREL | VK_G2,
90 VK_TPREL_G1 = VK_TPREL | VK_G1,
91 VK_TPREL_G1_NC = VK_TPREL | VK_G1 | VK_NC,
92 VK_TPREL_G0 = VK_TPREL | VK_G0,
93 VK_TPREL_G0_NC = VK_TPREL | VK_G0 | VK_NC,
94 VK_TPREL_HI12 = VK_TPREL | VK_HI12,
95 VK_TPREL_LO12 = VK_TPREL | VK_PAGEOFF,
96 VK_TPREL_LO12_NC = VK_TPREL | VK_PAGEOFF | VK_NC,
97 VK_TLSDESC_LO12 = VK_TLSDESC | VK_PAGEOFF,
98 VK_TLSDESC_PAGE = VK_TLSDESC | VK_PAGE,
99 VK_SECREL_LO12 = VK_SECREL | VK_PAGEOFF,
100 VK_SECREL_HI12 = VK_SECREL | VK_HI12,
101
102 VK_INVALID = 0xfff
103 };

122

136

142
143 /// @}
144

159
160 static bool classof(const MCExpr *E) {
161 return E->getKind() == MCExpr::Target;
162 }
163
164 static bool classof(const AArch64MCExpr *) { return true; }
165};
