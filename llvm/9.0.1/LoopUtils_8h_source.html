188/// @param OrigLoopID The loop ID of the loop before the transformation.
189/// @param FollowupAttrs List of attribute names that contain attributes to be
190/// added to the new loop ID.

218/// The mode sets how eager a transformation should be applied.
219enum TransformationMode {
220 /// The pass can use heuristics to determine whether a transformation should
221 /// be applied.
222 TM_Unspecified,
223
224 /// The transformation should be applied without considering a cost model.
225 TM_Enable,
226
227 /// The transformation should not be applied.
228 TM_Disable,
229
230 /// Force is a flag and should not be used alone.
231 TM_Force = 0x04,
232
233 /// The transformation was directed by the user, e.g. by a #pragma in
234 /// the source code. If the transformation could not be applied, a
235 /// warning should be emitted.
236 TM_ForcedByUser = TM_Enable | TM_Force,
237
238 /// The transformation must not be applied. For instance, `#pragma clang loop
239 /// unroll(disable)` explicitly forbids any unrolling to take place. Unlike
240 /// general loop metadata, it must not be dropped. Most passes should not
241 /// behave differently under TM_Disable and TM_SuppressedByUser.
242 TM_SuppressedByUser = TM_Disable | TM_Force
243};
