109// Helper function that transforms Inst (either an indirect-call instruction, or
110// an invoke instruction , to a conditional call to F. This is like:
111// if (Inst.CalledValue == F)
112// F(...);
113// else
114// Inst(...);
115// end
116// TotalCount is the profile count value that the instruction executes.
117// Count is the profile count value that F is the target function.
118// These two values are used to update the branch weight.
119// If \p AttachProfToDirectCall is true, a prof metadata is attached to the
120// new direct call to contain \p Count.
121// Returns the promoted direct call instruction.

188/// Calculate what to divide by to scale counts.
189///
190/// Given the maximum count, calculate a divisor that will scale all the
191/// weights to strictly less than std::numeric_limits<uint32_t>::max().
192static inline uint64_t calculateCountScale(uint64_t MaxCount) {
193 return MaxCount < std::numeric_limits<uint32_t>::max()
194 ? 1
195 : MaxCount / std::numeric_limits<uint32_t>::max() + 1;
196}
197
198/// Scale an individual branch count.
199///
200/// Scale a 64-bit weight down to 32-bits using \c Scale.
201///
202static inline uint32_t scaleBranchCount(uint64_t Count, uint64_t Scale) {
203 uint64_t Scaled = Count / Scale;
204 assert(Scaled <= std::numeric_limits<uint32_t>::max() && "overflow 32-bits");
205 return Scaled;
206}
