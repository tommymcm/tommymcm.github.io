201 /// Push a new Value to the LeaderTable onto the list for its value number.
202 void addToLeaderTable(uint32_t N, Value *V, const BasicBlock *BB) {
203 LeaderTableEntry &Curr = LeaderTable[N];
204 if (!Curr.Val) {
205 Curr.Val = V;
206 Curr.BB = BB;
207 return;
208 }
209
210 LeaderTableEntry *Node = TableAllocator.Allocate<LeaderTableEntry>();
211 Node->Val = V;
212 Node->BB = BB;
213 Node->Next = Curr.Next;
214 Curr.Next = Node;
215 }
216
217 /// Scan the list of values corresponding to a given
218 /// value number, and remove the given instruction if encountered.
219 void removeFromLeaderTable(uint32_t N, Instruction *I, BasicBlock *BB) {
220 LeaderTableEntry *Prev = nullptr;
221 LeaderTableEntry *Curr = &LeaderTable[N];
222
223 while (Curr && (Curr->Val != I || Curr->BB != BB)) {
224 Prev = Curr;
225 Curr = Curr->Next;
226 }
227
228 if (!Curr)
229 return;
230
231 if (Prev) {
232 Prev->Next = Curr->Next;
233 } else {
234 if (!Curr->Next) {
235 Curr->Val = nullptr;
236 Curr->BB = nullptr;
237 } else {
238 LeaderTableEntry *Next = Curr->Next;
239 Curr->Val = Next->Val;
240 Curr->BB = Next->BB;
241 Curr->Next = Next->Next;
242 }
243 }
244 }
