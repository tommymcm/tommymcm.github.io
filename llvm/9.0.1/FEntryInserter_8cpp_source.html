36bool FEntryInserter::runOnMachineFunction(MachineFunction &MF) {
37 const std::string FEntryName =
38 MF.getFunction().getFnAttribute("fentry-call").getValueAsString();
39 if (FEntryName != "true")
40 return false;
41
42 auto &FirstMBB = *MF.begin();
43 auto *TII = MF.getSubtarget().getInstrInfo();
44 BuildMI(FirstMBB, FirstMBB.begin(), DebugLoc(),
45 TII->get(TargetOpcode::FENTRY_CALL));
46 return true;
47}
