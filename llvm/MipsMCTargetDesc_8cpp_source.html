46/// Select the Mips CPU for the given triple and cpu name.
47/// FIXME: Merge with the copy in MipsSubtarget.cpp
48StringRef MIPS_MC::selectMipsCPU(const Triple &TT, StringRef CPU) {
49 if (CPU.empty() || CPU == "generic") {
50 if (TT.getSubArch() == llvm::Triple::MipsSubArch_r6) {
51 if (TT.isMIPS32())
52 CPU = "mips32r6";
53 else
54 CPU = "mips64r6";
55 } else {
56 if (TT.isMIPS32())
57 CPU = "mips32";
58 else
59 CPU = "mips64";
60 }
61 }
62 return CPU;
63}

83static MCAsmInfo *createMipsMCAsmInfo(const MCRegisterInfo &MRI,
84 const Triple &TT) {
85 MCAsmInfo *MAI = new MipsMCAsmInfo(TT);
86
87 unsigned SP = MRI.getDwarfRegNum(Mips::SP, true);
88 MCCFIInstruction Inst = MCCFIInstruction::createDefCfaRegister(nullptr, SP);
89 MAI->addInitialFrameState(Inst);
90
91 return MAI;
92}

136public:
137 MipsMCInstrAnalysis(const MCInstrInfo *Info) : MCInstrAnalysis(Info) {}

139 bool evaluateBranch(const MCInst &Inst, uint64_t Addr, uint64_t Size,
140 uint64_t &Target) const override {
141 unsigned NumOps = Inst.getNumOperands();
142 if (NumOps == 0)
143 return false;
144 switch (Info->get(Inst.getOpcode()).OpInfo[NumOps - 1].OperandType) {
145 case MCOI::OPERAND_UNKNOWN:
146 case MCOI::OPERAND_IMMEDIATE:
147 // jal, bal ...
148 Target = Inst.getOperand(NumOps - 1).getImm();
149 return true;
150 case MCOI::OPERAND_PCREL:
151 // b, j, beq ...
152 Target = Addr + Inst.getOperand(NumOps - 1).getImm();
153 return true;
154 default:
155 return false;
156 }
157 }
158};
159}

165extern "C" void LLVMInitializeMipsTargetMC() {
166 for (Target *T : {&getTheMipsTarget(), &getTheMipselTarget(),
167 &getTheMips64Target(), &getTheMips64elTarget()}) {
168 // Register the MC asm info.
169 RegisterMCAsmInfoFn X(*T, createMipsMCAsmInfo);
170
171 // Register the MC instruction info.
172 TargetRegistry::RegisterMCInstrInfo(*T, createMipsMCInstrInfo);
173
174 // Register the MC register info.
175 TargetRegistry::RegisterMCRegInfo(*T, createMipsMCRegisterInfo);
176
177 // Register the elf streamer.
178 TargetRegistry::RegisterELFStreamer(*T, createMCStreamer);
179
180 // Register the asm target streamer.
181 TargetRegistry::RegisterAsmTargetStreamer(*T, createMipsAsmTargetStreamer);
182
183 TargetRegistry::RegisterNullTargetStreamer(*T,
184 createMipsNullTargetStreamer);
185
186 // Register the MC subtarget info.
187 TargetRegistry::RegisterMCSubtargetInfo(*T, createMipsMCSubtargetInfo);
188
189 // Register the MC instruction analyzer.
190 TargetRegistry::RegisterMCInstrAnalysis(*T, createMipsMCInstrAnalysis);
191
192 // Register the MCInstPrinter.
193 TargetRegistry::RegisterMCInstPrinter(*T, createMipsMCInstPrinter);
194
195 TargetRegistry::RegisterObjectTargetStreamer(
196 *T, createMipsObjectTargetStreamer);
197
198 // Register the asm backend.
199 TargetRegistry::RegisterMCAsmBackend(*T, createMipsAsmBackend);
200 }
201
202 // Register the MC Code Emitter
203 for (Target *T : {&getTheMipsTarget(), &getTheMips64Target()})
204 TargetRegistry::RegisterMCCodeEmitter(*T, createMipsMCCodeEmitterEB);
205
206 for (Target *T : {&getTheMipselTarget(), &getTheMips64elTarget()})
207 TargetRegistry::RegisterMCCodeEmitter(*T, createMipsMCCodeEmitterEL);
208}
