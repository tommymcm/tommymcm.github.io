106static void printExpr(const MCExpr *Expr, const MCAsmInfo *MAI,
107 raw_ostream &OS) {
108 int Offset = 0;
109 const MCSymbolRefExpr *SRE;
110
111 if (const auto *CE = dyn_cast<MCConstantExpr>(Expr)) {
112 OS << "0x";
113 OS.write_hex(CE->getValue());
114 return;
115 }
116
117 if (const auto *BE = dyn_cast<MCBinaryExpr>(Expr)) {
118 SRE = dyn_cast<MCSymbolRefExpr>(BE->getLHS());
119 const auto *CE = dyn_cast<MCConstantExpr>(BE->getRHS());
120 assert(SRE && CE && "Binary expression must be sym+const.");
121 Offset = CE->getValue();
122 } else {
123 SRE = dyn_cast<MCSymbolRefExpr>(Expr);
124 assert(SRE && "Unexpected MCExpr type.");
125 }
126 assert(SRE->getKind() == MCSymbolRefExpr::VK_None);
127
128 // Symbols are prefixed with '@'
129 OS << '@';
130 SRE->getSymbol().print(OS, MAI);
131
132 if (Offset) {
133 if (Offset > 0)
134 OS << '+';
135 OS << Offset;
136 }
137}

139void ARCInstPrinter::printOperand(const MCInst *MI, unsigned OpNum,
140 raw_ostream &O) {
141 const MCOperand &Op = MI->getOperand(OpNum);
142 if (Op.isReg()) {
143 printRegName(O, Op.getReg());
144 return;
145 }
146
147 if (Op.isImm()) {
148 O << Op.getImm();
149 return;
150 }
151
152 assert(Op.isExpr() && "unknown operand kind in printOperand");
153 printExpr(Op.getExpr(), &MAI, O);
154}

156void ARCInstPrinter::printMemOperandRI(const MCInst *MI, unsigned OpNum,
157 raw_ostream &O) {
158 const MCOperand &base = MI->getOperand(OpNum);
159 const MCOperand &offset = MI->getOperand(OpNum + 1);
160 assert(base.isReg() && "Base should be register.");
161 assert(offset.isImm() && "Offset should be immediate.");
162 printRegName(O, base.getReg());
163 O << "," << offset.getImm();
164}
