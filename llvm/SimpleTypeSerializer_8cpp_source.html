6static void addPadding(BinaryStreamWriter &Writer) {
7 uint32_t Align = Writer.getOffset() % 4;

43 Prefix->RecordLen = Writer.getOffset() - sizeof(uint16_t);
44
45 return {ScratchBuffer.data(), Writer.getOffset()};
