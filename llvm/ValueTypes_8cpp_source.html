309/// Return the value type corresponding to the specified type. This returns all
310/// pointers as MVT::iPTR. If HandleUnknown is true, unknown types are returned
311/// as Other, otherwise they are invalid.
312MVT MVT::getVT(Type *Ty, bool HandleUnknown){
313 switch (Ty->getTypeID()) {
314 default:
315 if (HandleUnknown) return MVT(MVT::Other);
316 llvm_unreachable("Unknown type!");
317 case Type::VoidTyID:
318 return MVT::isVoid;
319 case Type::IntegerTyID:
320 return getIntegerVT(cast<IntegerType>(Ty)->getBitWidth());
321 case Type::HalfTyID: return MVT(MVT::f16);
322 case Type::FloatTyID: return MVT(MVT::f32);
323 case Type::DoubleTyID: return MVT(MVT::f64);
324 case Type::X86_FP80TyID: return MVT(MVT::f80);
325 case Type::X86_MMXTyID: return MVT(MVT::x86mmx);
326 case Type::FP128TyID: return MVT(MVT::f128);
327 case Type::PPC_FP128TyID: return MVT(MVT::ppcf128);
328 case Type::PointerTyID: return MVT(MVT::iPTR);
329 case Type::VectorTyID: {
330 VectorType *VTy = cast<VectorType>(Ty);
331 return getVectorVT(
332 getVT(VTy->getElementType(), false), VTy->getNumElements());
333 }
334 }
335}

337/// getEVT - Return the value type corresponding to the specified type. This
338/// returns all pointers as MVT::iPTR. If HandleUnknown is true, unknown types
339/// are returned as Other, otherwise they are invalid.
340EVT EVT::getEVT(Type *Ty, bool HandleUnknown){
341 switch (Ty->getTypeID()) {
342 default:
343 return MVT::getVT(Ty, HandleUnknown);
344 case Type::IntegerTyID:
345 return getIntegerVT(Ty->getContext(), cast<IntegerType>(Ty)->getBitWidth());
346 case Type::VectorTyID: {
347 VectorType *VTy = cast<VectorType>(Ty);
348 return getVectorVT(Ty->getContext(), getEVT(VTy->getElementType(), false),
349 VTy->getNumElements());
350 }
351 }
