69void Mips16InstrInfo::copyPhysReg(MachineBasicBlock &MBB,
70 MachineBasicBlock::iterator I,
71 const DebugLoc &DL, unsigned DestReg,
72 unsigned SrcReg, bool KillSrc) const {
73 unsigned Opc = 0;
74
75 if (Mips::CPU16RegsRegClass.contains(DestReg) &&
76 Mips::GPR32RegClass.contains(SrcReg))
77 Opc = Mips::MoveR3216;
78 else if (Mips::GPR32RegClass.contains(DestReg) &&
79 Mips::CPU16RegsRegClass.contains(SrcReg))
80 Opc = Mips::Move32R16;
81 else if ((SrcReg == Mips::HI0) &&
82 (Mips::CPU16RegsRegClass.contains(DestReg)))
83 Opc = Mips::Mfhi16, SrcReg = 0;
84 else if ((SrcReg == Mips::LO0) &&
85 (Mips::CPU16RegsRegClass.contains(DestReg)))
86 Opc = Mips::Mflo16, SrcReg = 0;
87
88 assert(Opc && "Cannot copy registers");
89
90 MachineInstrBuilder MIB = BuildMI(MBB, I, DL, get(Opc));
91
92 if (DestReg)
93 MIB.addReg(DestReg, RegState::Define);
94
95 if (SrcReg)
96 MIB.addReg(SrcReg, getKillRegState(KillSrc));
97}

146bool Mips16InstrInfo::expandPostRAPseudo(MachineInstr &MI) const {
147 MachineBasicBlock &MBB = *MI.getParent();
148 switch (MI.getDesc().getOpcode()) {
149 default:
150 return false;
151 case Mips::RetRA16:
152 ExpandRetRA16(MBB, MI, Mips::JrcRa16);

154 }
155
156 MBB.erase(MI.getIterator());

188static void addSaveRestoreRegs(MachineInstrBuilder &MIB,
189 const std::vector<CalleeSavedInfo> &CSI,
190 unsigned Flags = 0) {
191 for (unsigned i = 0, e = CSI.size(); i != e; ++i) {
192 // Add the callee-saved register as live-in. Do not add if the register is
193 // RA and return address is taken, because it has already been added in
194 // method MipsTargetLowering::lowerRETURNADDR.
195 // It's killed at the spill, unless the register is RA and return address
196 // is taken.
197 unsigned Reg = CSI[e-i-1].getReg();
198 switch (Reg) {
199 case Mips::RA:
200 case Mips::S0:
201 case Mips::S1:
202 MIB.addReg(Reg, Flags);
203 break;
204 case Mips::S2:
205 break;
206 default:
207 llvm_unreachable("unexpected mips16 callee saved register");
208
209 }
210 }
211}

213// Adjust SP by FrameSize bytes. Save RA, S0, S1
214void Mips16InstrInfo::makeFrame(unsigned SP, int64_t FrameSize,
215 MachineBasicBlock &MBB,
216 MachineBasicBlock::iterator I) const {
217 DebugLoc DL;
218 MachineFunction &MF = *MBB.getParent();
219 MachineFrameInfo &MFI = MF.getFrameInfo();
220 const BitVector Reserved = RI.getReservedRegs(MF);
221 bool SaveS2 = Reserved[Mips::S2];
222 MachineInstrBuilder MIB;
223 unsigned Opc = ((FrameSize <= 128) && !SaveS2)? Mips::Save16:Mips::SaveX16;
224 MIB = BuildMI(MBB, I, DL, get(Opc));
225 const std::vector<CalleeSavedInfo> &CSI = MFI.getCalleeSavedInfo();
226 addSaveRestoreRegs(MIB, CSI);
227 if (SaveS2)
228 MIB.addReg(Mips::S2);
229 if (isUInt<11>(FrameSize))
230 MIB.addImm(FrameSize);
231 else {
232 int Base = 2040; // should create template function like isUInt that
233 // returns largest possible n bit unsigned integer
234 int64_t Remainder = FrameSize - Base;
235 MIB.addImm(Base);
236 if (isInt<16>(-Remainder))
237 BuildAddiuSpImm(MBB, I, -Remainder);
238 else
239 adjustStackPtrBig(SP, -Remainder, MBB, I, Mips::V0, Mips::V1);
240 }
241}

243// Adjust SP by FrameSize bytes. Restore RA, S0, S1
244void Mips16InstrInfo::restoreFrame(unsigned SP, int64_t FrameSize,
245 MachineBasicBlock &MBB,
246 MachineBasicBlock::iterator I) const {
247 DebugLoc DL = I != MBB.end() ? I->getDebugLoc() : DebugLoc();
248 MachineFunction *MF = MBB.getParent();
249 MachineFrameInfo &MFI = MF->getFrameInfo();
250 const BitVector Reserved = RI.getReservedRegs(*MF);
251 bool SaveS2 = Reserved[Mips::S2];
252 MachineInstrBuilder MIB;
253 unsigned Opc = ((FrameSize <= 128) && !SaveS2)?
254 Mips::Restore16:Mips::RestoreX16;
255
256 if (!isUInt<11>(FrameSize)) {
257 unsigned Base = 2040;
258 int64_t Remainder = FrameSize - Base;
259 FrameSize = Base; // should create template function like isUInt that
260 // returns largest possible n bit unsigned integer
261
262 if (isInt<16>(Remainder))
263 BuildAddiuSpImm(MBB, I, Remainder);
264 else
265 adjustStackPtrBig(SP, Remainder, MBB, I, Mips::A0, Mips::A1);
266 }
267 MIB = BuildMI(MBB, I, DL, get(Opc));
268 const std::vector<CalleeSavedInfo> &CSI = MFI.getCalleeSavedInfo();
269 addSaveRestoreRegs(MIB, CSI, RegState::Define);
270 if (SaveS2)
271 MIB.addReg(Mips::S2, RegState::Define);
272 MIB.addImm(FrameSize);
273}

275// Adjust SP by Amount bytes where bytes can be up to 32bit number.
276// This can only be called at times that we know that there is at least one free
277// register.
278// This is clearly safe at prologue and epilogue.
279void Mips16InstrInfo::adjustStackPtrBig(unsigned SP, int64_t Amount,
280 MachineBasicBlock &MBB,
281 MachineBasicBlock::iterator I,
282 unsigned Reg1, unsigned Reg2) const {
283 DebugLoc DL;
284 //
285 // li reg1, constant
286 // move reg2, sp
287 // add reg1, reg1, reg2
288 // move sp, reg1
289 //
290 //
291 MachineInstrBuilder MIB1 = BuildMI(MBB, I, DL, get(Mips::LwConstant32), Reg1);
292 MIB1.addImm(Amount).addImm(-1);
293 MachineInstrBuilder MIB2 = BuildMI(MBB, I, DL, get(Mips::MoveR3216), Reg2);
294 MIB2.addReg(Mips::SP, RegState::Kill);
295 MachineInstrBuilder MIB3 = BuildMI(MBB, I, DL, get(Mips::AdduRxRyRz16), Reg1);
296 MIB3.addReg(Reg1);
297 MIB3.addReg(Reg2, RegState::Kill);
298 MachineInstrBuilder MIB4 = BuildMI(MBB, I, DL, get(Mips::Move32R16),
299 Mips::SP);
300 MIB4.addReg(Reg1, RegState::Kill);
301}

322/// This function generates the sequence of instructions needed to get the
323/// result of adding register REG and immediate IMM.
324unsigned Mips16InstrInfo::loadImmediate(unsigned FrameReg, int64_t Imm,
325 MachineBasicBlock &MBB,
326 MachineBasicBlock::iterator II,
327 const DebugLoc &DL,
328 unsigned &NewImm) const {
329 //
330 // given original instruction is:
331 // Instr rx, T[offset] where offset is too big.
332 //
333 // lo = offset & 0xFFFF
334 // hi = ((offset >> 16) + (lo >> 15)) & 0xFFFF;
335 //
336 // let T = temporary register
337 // li T, hi
338 // shl T, 16
339 // add T, Rx, T
340 //
341 RegScavenger rs;
342 int32_t lo = Imm & 0xFFFF;
343 NewImm = lo;
344 int Reg =0;
345 int SpReg = 0;
346
347 rs.enterBasicBlock(MBB);
348 rs.forward(II);
349 //
350 // We need to know which registers can be used, in the case where there
351 // are not enough free registers. We exclude all registers that
352 // are used in the instruction that we are helping.
353 // // Consider all allocatable registers in the register class initially
354 BitVector Candidates =
355 RI.getAllocatableSet
356 (*II->getParent()->getParent(), &Mips::CPU16RegsRegClass);
357 // Exclude all the registers being used by the instruction.
358 for (unsigned i = 0, e = II->getNumOperands(); i != e; ++i) {
359 MachineOperand &MO = II->getOperand(i);
360 if (MO.isReg() && MO.getReg() != 0 && !MO.isDef() &&
361 !TargetRegisterInfo::isVirtualRegister(MO.getReg()))
362 Candidates.reset(MO.getReg());
363 }
364
365 // If the same register was used and defined in an instruction, then
366 // it will not be in the list of candidates.
367 //
368 // we need to analyze the instruction that we are helping.
369 // we need to know if it defines register x but register x is not
370 // present as an operand of the instruction. this tells
371 // whether the register is live before the instruction. if it's not
372 // then we don't need to save it in case there are no free registers.
373 int DefReg = 0;
374 for (unsigned i = 0, e = II->getNumOperands(); i != e; ++i) {
375 MachineOperand &MO = II->getOperand(i);
376 if (MO.isReg() && MO.isDef()) {
377 DefReg = MO.getReg();
378 break;
379 }
380 }
381
382 BitVector Available = rs.getRegsAvailable(&Mips::CPU16RegsRegClass);
383 Available &= Candidates;
384 //
385 // we use T0 for the first register, if we need to save something away.
386 // we use T1 for the second register, if we need to save something away.
387 //
388 unsigned FirstRegSaved =0, SecondRegSaved=0;
389 unsigned FirstRegSavedTo = 0, SecondRegSavedTo = 0;
390
391 Reg = Available.find_first();
392
393 if (Reg == -1) {
394 Reg = Candidates.find_first();
395 Candidates.reset(Reg);
396 if (DefReg != Reg) {
397 FirstRegSaved = Reg;
398 FirstRegSavedTo = Mips::T0;
399 copyPhysReg(MBB, II, DL, FirstRegSavedTo, FirstRegSaved, true);
400 }
401 }
402 else
403 Available.reset(Reg);
404 BuildMI(MBB, II, DL, get(Mips::LwConstant32), Reg).addImm(Imm).addImm(-1);
405 NewImm = 0;
406 if (FrameReg == Mips::SP) {
407 SpReg = Available.find_first();
408 if (SpReg == -1) {
409 SpReg = Candidates.find_first();
410 // Candidates.reset(SpReg); // not really needed
411 if (DefReg!= SpReg) {
412 SecondRegSaved = SpReg;
413 SecondRegSavedTo = Mips::T1;
414 }
415 if (SecondRegSaved)
416 copyPhysReg(MBB, II, DL, SecondRegSavedTo, SecondRegSaved, true);
417 }
418 else
419 Available.reset(SpReg);
420 copyPhysReg(MBB, II, DL, SpReg, Mips::SP, false);
421 BuildMI(MBB, II, DL, get(Mips:: AdduRxRyRz16), Reg).addReg(SpReg, RegState::Kill)
422 .addReg(Reg);
423 }
424 else
425 BuildMI(MBB, II, DL, get(Mips:: AdduRxRyRz16), Reg).addReg(FrameReg)
426 .addReg(Reg, RegState::Kill);
427 if (FirstRegSaved || SecondRegSaved) {
428 II = std::next(II);
429 if (FirstRegSaved)
430 copyPhysReg(MBB, II, DL, FirstRegSaved, FirstRegSavedTo, true);
431 if (SecondRegSaved)
432 copyPhysReg(MBB, II, DL, SecondRegSaved, SecondRegSavedTo, true);
433 }
434 return Reg;
435}

475bool Mips16InstrInfo::validImmediate(unsigned Opcode, unsigned Reg,
476 int64_t Amount) {
477 switch (Opcode) {
478 case Mips::LbRxRyOffMemX16:
479 case Mips::LbuRxRyOffMemX16:
480 case Mips::LhRxRyOffMemX16:
481 case Mips::LhuRxRyOffMemX16:
482 case Mips::SbRxRyOffMemX16:
483 case Mips::ShRxRyOffMemX16:
484 case Mips::LwRxRyOffMemX16:
485 case Mips::SwRxRyOffMemX16:
486 case Mips::SwRxSpImmX16:
487 case Mips::LwRxSpImmX16:
488 return isInt<16>(Amount);
489 case Mips::AddiuRxRyOffMemX16:
490 if ((Reg == Mips::PC) || (Reg == Mips::SP))
491 return isInt<16>(Amount);
492 return isInt<15>(Amount);
493 }
494 llvm_unreachable("unexpected Opcode in validImmediate");
495}
