54static MCAsmInfo *createARCMCAsmInfo(const MCRegisterInfo &MRI,
55 const Triple &TT) {
56 MCAsmInfo *MAI = new ARCMCAsmInfo(TT);
57
58 // Initial state of the frame pointer is SP.
59 MCCFIInstruction Inst = MCCFIInstruction::createDefCfa(nullptr, ARC::SP, 0);
60 MAI->addInitialFrameState(Inst);
61
62 return MAI;
63}
