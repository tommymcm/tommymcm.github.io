42void MCExpr::print(raw_ostream &OS, const MCAsmInfo *MAI, bool InParens) const {
43 switch (getKind()) {
44 case MCExpr::Target:
45 return cast<MCTargetExpr>(this)->printImpl(OS, MAI);
46 case MCExpr::Constant: {
47 auto Value = cast<MCConstantExpr>(*this).getValue();
48 auto PrintInHex = cast<MCConstantExpr>(*this).useHexFormat();
49 if (PrintInHex)
50 OS << "0x" << Twine::utohexstr(Value);
51 else
52 OS << Value;
53 return;
54 }
55 case MCExpr::SymbolRef: {
56 const MCSymbolRefExpr &SRE = cast<MCSymbolRefExpr>(*this);
57 const MCSymbol &Sym = SRE.getSymbol();
58 // Parenthesize names that start with $ so that they don't look like
59 // absolute names.
60 bool UseParens =
61 !InParens && !Sym.getName().empty() && Sym.getName()[0] == '$';
62 if (UseParens) {
63 OS << '(';
64 Sym.print(OS, MAI);
65 OS << ')';
66 } else
67 Sym.print(OS, MAI);
68
69 if (SRE.getKind() != MCSymbolRefExpr::VK_None)
70 SRE.printVariantKind(OS);
71
72 return;
73 }
74
75 case MCExpr::Unary: {
76 const MCUnaryExpr &UE = cast<MCUnaryExpr>(*this);
77 switch (UE.getOpcode()) {
78 case MCUnaryExpr::LNot: OS << '!'; break;
79 case MCUnaryExpr::Minus: OS << '-'; break;
80 case MCUnaryExpr::Not: OS << '~'; break;
81 case MCUnaryExpr::Plus: OS << '+'; break;
82 }
83 bool Binary = UE.getSubExpr()->getKind() == MCExpr::Binary;
84 if (Binary) OS << "(";
85 UE.getSubExpr()->print(OS, MAI);
86 if (Binary) OS << ")";
87 return;
88 }
89
90 case MCExpr::Binary: {
91 const MCBinaryExpr &BE = cast<MCBinaryExpr>(*this);
92
93 // Only print parens around the LHS if it is non-trivial.
94 if (isa<MCConstantExpr>(BE.getLHS()) || isa<MCSymbolRefExpr>(BE.getLHS())) {
95 BE.getLHS()->print(OS, MAI);
96 } else {
97 OS << '(';
98 BE.getLHS()->print(OS, MAI);
99 OS << ')';
100 }
101
102 switch (BE.getOpcode()) {
103 case MCBinaryExpr::Add:
104 // Print "X-42" instead of "X+-42".
105 if (const MCConstantExpr *RHSC = dyn_cast<MCConstantExpr>(BE.getRHS())) {
106 if (RHSC->getValue() < 0) {
107 OS << RHSC->getValue();
108 return;
109 }
110 }
111
112 OS << '+';
113 break;
114 case MCBinaryExpr::AShr: OS << ">>"; break;
115 case MCBinaryExpr::And: OS << '&'; break;
116 case MCBinaryExpr::Div: OS << '/'; break;
117 case MCBinaryExpr::EQ: OS << "=="; break;
118 case MCBinaryExpr::GT: OS << '>'; break;
119 case MCBinaryExpr::GTE: OS << ">="; break;
120 case MCBinaryExpr::LAnd: OS << "&&"; break;
121 case MCBinaryExpr::LOr: OS << "||"; break;
122 case MCBinaryExpr::LShr: OS << ">>"; break;
123 case MCBinaryExpr::LT: OS << '<'; break;
124 case MCBinaryExpr::LTE: OS << "<="; break;
125 case MCBinaryExpr::Mod: OS << '%'; break;
126 case MCBinaryExpr::Mul: OS << '*'; break;
127 case MCBinaryExpr::NE: OS << "!="; break;
128 case MCBinaryExpr::Or: OS << '|'; break;
129 case MCBinaryExpr::Shl: OS << "<<"; break;
130 case MCBinaryExpr::Sub: OS << '-'; break;
131 case MCBinaryExpr::Xor: OS << '^'; break;
132 }
133
134 // Only print parens around the LHS if it is non-trivial.
135 if (isa<MCConstantExpr>(BE.getRHS()) || isa<MCSymbolRefExpr>(BE.getRHS())) {
136 BE.getRHS()->print(OS, MAI);
137 } else {
138 OS << '(';
139 BE.getRHS()->print(OS, MAI);
140 OS << ')';
141 }
142 return;
143 }
144 }
145
146 llvm_unreachable("Invalid expression kind!");
147}

484bool MCExpr::evaluateAsAbsolute(int64_t &Res, const MCAssembler *Asm,
485 const MCAsmLayout *Layout,
486 const SectionAddrMap *Addrs) const {
487 // FIXME: The use if InSet = Addrs is a hack. Setting InSet causes us
488 // absolutize differences across sections and that is what the MachO writer
489 // uses Addrs for.
490 return evaluateAsAbsolute(Res, Asm, Layout, Addrs, Addrs);
491}
492
493bool MCExpr::evaluateAsAbsolute(int64_t &Res, const MCAssembler *Asm,
494 const MCAsmLayout *Layout,
495 const SectionAddrMap *Addrs, bool InSet) const {
496 MCValue Value;
497
498 // Fast path constants.
499 if (const MCConstantExpr *CE = dyn_cast<MCConstantExpr>(this)) {
500 Res = CE->getValue();
501 return true;
502 }
503
504 bool IsRelocatable =
505 evaluateAsRelocatableImpl(Value, Asm, Layout, nullptr, Addrs, InSet);
506
507 // Record the current value.
508 Res = Value.getConstant();
509
510 return IsRelocatable && Value.isAbsolute();
511}

513/// Helper method for \see EvaluateSymbolAdd().
514static void AttemptToFoldSymbolOffsetDifference(
515 const MCAssembler *Asm, const MCAsmLayout *Layout,
516 const SectionAddrMap *Addrs, bool InSet, const MCSymbolRefExpr *&A,
517 const MCSymbolRefExpr *&B, int64_t &Addend) {
518 if (!A || !B)
519 return;
520
521 const MCSymbol &SA = A->getSymbol();
522 const MCSymbol &SB = B->getSymbol();
523
524 if (SA.isUndefined() || SB.isUndefined())
525 return;
526
527 if (!Asm->getWriter().isSymbolRefDifferenceFullyResolved(*Asm, A, B, InSet))
528 return;
529
530 if (SA.getFragment() == SB.getFragment() && !SA.isVariable() &&
531 !SA.isUnset() && !SB.isVariable() && !SB.isUnset()) {
532 Addend += (SA.getOffset() - SB.getOffset());
533
534 // Pointers to Thumb symbols need to have their low-bit set to allow
535 // for interworking.
536 if (Asm->isThumbFunc(&SA))
537 Addend |= 1;
538
539 // If symbol is labeled as micromips, we set low-bit to ensure
540 // correct offset in .gcc_except_table
541 if (Asm->getBackend().isMicroMips(&SA))
542 Addend |= 1;
543
544 // Clear the symbol expr pointers to indicate we have folded these
545 // operands.
546 A = B = nullptr;
547 return;
548 }
549
550 if (!Layout)
551 return;
552
553 const MCSection &SecA = *SA.getFragment()->getParent();
554 const MCSection &SecB = *SB.getFragment()->getParent();
555
556 if ((&SecA != &SecB) && !Addrs)
557 return;
558
559 // Eagerly evaluate.
560 Addend += Layout->getSymbolOffset(A->getSymbol()) -
561 Layout->getSymbolOffset(B->getSymbol());
562 if (Addrs && (&SecA != &SecB))
563 Addend += (Addrs->lookup(&SecA) - Addrs->lookup(&SecB));
564
565 // Pointers to Thumb symbols need to have their low-bit set to allow
566 // for interworking.
567 if (Asm->isThumbFunc(&SA))
568 Addend |= 1;
569
570 // If symbol is labeled as micromips, we set low-bit to ensure
571 // correct offset in .gcc_except_table
572 if (Asm->getBackend().isMicroMips(&SA))
573 Addend |= 1;
574
575 // Clear the symbol expr pointers to indicate we have folded these
576 // operands.
577 A = B = nullptr;
578}

580/// Evaluate the result of an add between (conceptually) two MCValues.
581///
582/// This routine conceptually attempts to construct an MCValue:
583/// Result = (Result_A - Result_B + Result_Cst)
584/// from two MCValue's LHS and RHS where
585/// Result = LHS + RHS
586/// and
587/// Result = (LHS_A - LHS_B + LHS_Cst) + (RHS_A - RHS_B + RHS_Cst).
588///
589/// This routine attempts to aggresively fold the operands such that the result
590/// is representable in an MCValue, but may not always succeed.
591///
592/// \returns True on success, false if the result is not representable in an
593/// MCValue.
594
595/// NOTE: It is really important to have both the Asm and Layout arguments.
596/// They might look redundant, but this function can be used before layout
597/// is done (see the object streamer for example) and having the Asm argument
598/// lets us avoid relaxations early.
599static bool
600EvaluateSymbolicAdd(const MCAssembler *Asm, const MCAsmLayout *Layout,
601 const SectionAddrMap *Addrs, bool InSet, const MCValue &LHS,
602 const MCSymbolRefExpr *RHS_A, const MCSymbolRefExpr *RHS_B,
603 int64_t RHS_Cst, MCValue &Res) {
604 // FIXME: This routine (and other evaluation parts) are *incredibly* sloppy
605 // about dealing with modifiers. This will ultimately bite us, one day.
606 const MCSymbolRefExpr *LHS_A = LHS.getSymA();
607 const MCSymbolRefExpr *LHS_B = LHS.getSymB();
608 int64_t LHS_Cst = LHS.getConstant();
609
610 // Fold the result constant immediately.
611 int64_t Result_Cst = LHS_Cst + RHS_Cst;
612
613 assert((!Layout || Asm) &&
614 "Must have an assembler object if layout is given!");
615
616 // If we have a layout, we can fold resolved differences. Do not do this if
617 // the backend requires this to be emitted as individual relocations, unless
618 // the InSet flag is set to get the current difference anyway (used for
619 // example to calculate symbol sizes).
620 if (Asm &&
621 (InSet || !Asm->getBackend().requiresDiffExpressionRelocations())) {
622 // First, fold out any differences which are fully resolved. By
623 // reassociating terms in
624 // Result = (LHS_A - LHS_B + LHS_Cst) + (RHS_A - RHS_B + RHS_Cst).
625 // we have the four possible differences:
626 // (LHS_A - LHS_B),
627 // (LHS_A - RHS_B),
628 // (RHS_A - LHS_B),
629 // (RHS_A - RHS_B).
630 // Since we are attempting to be as aggressive as possible about folding, we
631 // attempt to evaluate each possible alternative.
632 AttemptToFoldSymbolOffsetDifference(Asm, Layout, Addrs, InSet, LHS_A, LHS_B,
633 Result_Cst);
634 AttemptToFoldSymbolOffsetDifference(Asm, Layout, Addrs, InSet, LHS_A, RHS_B,
635 Result_Cst);
636 AttemptToFoldSymbolOffsetDifference(Asm, Layout, Addrs, InSet, RHS_A, LHS_B,
637 Result_Cst);
638 AttemptToFoldSymbolOffsetDifference(Asm, Layout, Addrs, InSet, RHS_A, RHS_B,
639 Result_Cst);
640 }
641
642 // We can't represent the addition or subtraction of two symbols.
643 if ((LHS_A && RHS_A) || (LHS_B && RHS_B))
644 return false;
645
646 // At this point, we have at most one additive symbol and one subtractive
647 // symbol -- find them.
648 const MCSymbolRefExpr *A = LHS_A ? LHS_A : RHS_A;
649 const MCSymbolRefExpr *B = LHS_B ? LHS_B : RHS_B;
650
651 Res = MCValue::get(A, B, Result_Cst);
652 return true;
653}

682bool MCExpr::evaluateAsRelocatableImpl(MCValue &Res, const MCAssembler *Asm,
683 const MCAsmLayout *Layout,
684 const MCFixup *Fixup,
685 const SectionAddrMap *Addrs,
686 bool InSet) const {
687 ++stats::MCExprEvaluate;
688
689 switch (getKind()) {
690 case Target:
691 return cast<MCTargetExpr>(this)->evaluateAsRelocatableImpl(Res, Layout,
692 Fixup);
693
694 case Constant:
695 Res = MCValue::get(cast<MCConstantExpr>(this)->getValue());
696 return true;
697
698 case SymbolRef: {
699 const MCSymbolRefExpr *SRE = cast<MCSymbolRefExpr>(this);
700 const MCSymbol &Sym = SRE->getSymbol();
701
702 // Evaluate recursively if this is a variable.
703 if (Sym.isVariable() && SRE->getKind() == MCSymbolRefExpr::VK_None &&
704 canExpand(Sym, InSet)) {
705 bool IsMachO = SRE->hasSubsectionsViaSymbols();
706 if (Sym.getVariableValue()->evaluateAsRelocatableImpl(
707 Res, Asm, Layout, Fixup, Addrs, InSet || IsMachO)) {
708 if (!IsMachO)
709 return true;
710
711 const MCSymbolRefExpr *A = Res.getSymA();
712 const MCSymbolRefExpr *B = Res.getSymB();
713 // FIXME: This is small hack. Given
714 // a = b + 4
715 // .long a
716 // the OS X assembler will completely drop the 4. We should probably
717 // include it in the relocation or produce an error if that is not
718 // possible.
719 // Allow constant expressions.
720 if (!A && !B)
721 return true;
722 // Allows aliases with zero offset.
723 if (Res.getConstant() == 0 && (!A || !B))
724 return true;
725 }
726 }
727
728 Res = MCValue::get(SRE, nullptr, 0);
729 return true;
730 }
731
732 case Unary: {
733 const MCUnaryExpr *AUE = cast<MCUnaryExpr>(this);
734 MCValue Value;
735
736 if (!AUE->getSubExpr()->evaluateAsRelocatableImpl(Value, Asm, Layout, Fixup,
737 Addrs, InSet))
738 return false;
739
740 switch (AUE->getOpcode()) {
741 case MCUnaryExpr::LNot:
742 if (!Value.isAbsolute())
743 return false;
744 Res = MCValue::get(!Value.getConstant());
745 break;
746 case MCUnaryExpr::Minus:
747 /// -(a - b + const) ==> (b - a - const)
748 if (Value.getSymA() && !Value.getSymB())
749 return false;
750
751 // The cast avoids undefined behavior if the constant is INT64_MIN.
752 Res = MCValue::get(Value.getSymB(), Value.getSymA(),
753 -(uint64_t)Value.getConstant());
754 break;
755 case MCUnaryExpr::Not:
756 if (!Value.isAbsolute())
757 return false;
758 Res = MCValue::get(~Value.getConstant());
759 break;
760 case MCUnaryExpr::Plus:
761 Res = Value;
762 break;
763 }
764
765 return true;
766 }
767
768 case Binary: {
769 const MCBinaryExpr *ABE = cast<MCBinaryExpr>(this);
770 MCValue LHSValue, RHSValue;
771
772 if (!ABE->getLHS()->evaluateAsRelocatableImpl(LHSValue, Asm, Layout, Fixup,
773 Addrs, InSet) ||
774 !ABE->getRHS()->evaluateAsRelocatableImpl(RHSValue, Asm, Layout, Fixup,
775 Addrs, InSet)) {
776 // Check if both are Target Expressions, see if we can compare them.
777 if (const MCTargetExpr *L = dyn_cast<MCTargetExpr>(ABE->getLHS()))
778 if (const MCTargetExpr *R = cast<MCTargetExpr>(ABE->getRHS())) {
779 switch (ABE->getOpcode()) {
780 case MCBinaryExpr::EQ:
781 Res = MCValue::get((L->isEqualTo(R)) ? -1 : 0);
782 return true;
783 case MCBinaryExpr::NE:
784 Res = MCValue::get((R->isEqualTo(R)) ? 0 : -1);
785 return true;
786 default: break;
787 }
788 }
789 return false;
790 }
791
792 // We only support a few operations on non-constant expressions, handle
793 // those first.
794 if (!LHSValue.isAbsolute() || !RHSValue.isAbsolute()) {
795 switch (ABE->getOpcode()) {
796 default:
797 return false;
798 case MCBinaryExpr::Sub:
799 // Negate RHS and add.
800 // The cast avoids undefined behavior if the constant is INT64_MIN.
801 return EvaluateSymbolicAdd(Asm, Layout, Addrs, InSet, LHSValue,
802 RHSValue.getSymB(), RHSValue.getSymA(),
803 -(uint64_t)RHSValue.getConstant(), Res);
804
805 case MCBinaryExpr::Add:
806 return EvaluateSymbolicAdd(Asm, Layout, Addrs, InSet, LHSValue,
807 RHSValue.getSymA(), RHSValue.getSymB(),
808 RHSValue.getConstant(), Res);
809 }
810 }
811
812 // FIXME: We need target hooks for the evaluation. It may be limited in
813 // width, and gas defines the result of comparisons differently from
814 // Apple as.
815 int64_t LHS = LHSValue.getConstant(), RHS = RHSValue.getConstant();
816 int64_t Result = 0;
817 auto Op = ABE->getOpcode();
818 switch (Op) {
819 case MCBinaryExpr::AShr: Result = LHS >> RHS; break;
820 case MCBinaryExpr::Add: Result = LHS + RHS; break;
821 case MCBinaryExpr::And: Result = LHS & RHS; break;
822 case MCBinaryExpr::Div:
823 case MCBinaryExpr::Mod:
824 // Handle division by zero. gas just emits a warning and keeps going,
825 // we try to be stricter.
826 // FIXME: Currently the caller of this function has no way to understand
827 // we're bailing out because of 'division by zero'. Therefore, it will
828 // emit a 'expected relocatable expression' error. It would be nice to
829 // change this code to emit a better diagnostic.
830 if (RHS == 0)
831 return false;
832 if (ABE->getOpcode() == MCBinaryExpr::Div)
833 Result = LHS / RHS;
834 else
835 Result = LHS % RHS;
836 break;
837 case MCBinaryExpr::EQ: Result = LHS == RHS; break;
838 case MCBinaryExpr::GT: Result = LHS > RHS; break;
839 case MCBinaryExpr::GTE: Result = LHS >= RHS; break;
840 case MCBinaryExpr::LAnd: Result = LHS && RHS; break;
841 case MCBinaryExpr::LOr: Result = LHS || RHS; break;
842 case MCBinaryExpr::LShr: Result = uint64_t(LHS) >> uint64_t(RHS); break;
843 case MCBinaryExpr::LT: Result = LHS < RHS; break;
844 case MCBinaryExpr::LTE: Result = LHS <= RHS; break;
845 case MCBinaryExpr::Mul: Result = LHS * RHS; break;
846 case MCBinaryExpr::NE: Result = LHS != RHS; break;
847 case MCBinaryExpr::Or: Result = LHS | RHS; break;
848 case MCBinaryExpr::Shl: Result = uint64_t(LHS) << uint64_t(RHS); break;
849 case MCBinaryExpr::Sub: Result = LHS - RHS; break;
850 case MCBinaryExpr::Xor: Result = LHS ^ RHS; break;
851 }
852
853 switch (Op) {
854 default:
855 Res = MCValue::get(Result);
856 break;
857 case MCBinaryExpr::EQ:
858 case MCBinaryExpr::GT:
859 case MCBinaryExpr::GTE:
860 case MCBinaryExpr::LT:
861 case MCBinaryExpr::LTE:
862 case MCBinaryExpr::NE:
863 // A comparison operator returns a -1 if true and 0 if false.
864 Res = MCValue::get(Result ? -1 : 0);
865 break;
866 }
867
868 return true;
869 }
870 }
871
872 llvm_unreachable("Invalid assembly expression kind!");
873}

875MCFragment *MCExpr::findAssociatedFragment() const {
876 switch (getKind()) {
877 case Target:
878 // We never look through target specific expressions.
879 return cast<MCTargetExpr>(this)->findAssociatedFragment();
880
881 case Constant:
882 return MCSymbol::AbsolutePseudoFragment;
883
884 case SymbolRef: {
885 const MCSymbolRefExpr *SRE = cast<MCSymbolRefExpr>(this);
886 const MCSymbol &Sym = SRE->getSymbol();
887 return Sym.getFragment();
888 }
889
890 case Unary:
891 return cast<MCUnaryExpr>(this)->getSubExpr()->findAssociatedFragment();
892
893 case Binary: {
894 const MCBinaryExpr *BE = cast<MCBinaryExpr>(this);
895 MCFragment *LHS_F = BE->getLHS()->findAssociatedFragment();
896 MCFragment *RHS_F = BE->getRHS()->findAssociatedFragment();
897
898 // If either is absolute, return the other.
899 if (LHS_F == MCSymbol::AbsolutePseudoFragment)
900 return RHS_F;
901 if (RHS_F == MCSymbol::AbsolutePseudoFragment)
902 return LHS_F;
903
904 // Not always correct, but probably the best we can do without more context.
905 if (BE->getOpcode() == MCBinaryExpr::Sub)
906 return MCSymbol::AbsolutePseudoFragment;
907
908 // Otherwise, return the first non-null fragment.
909 return LHS_F ? LHS_F : RHS_F;
910 }
911 }
912
913 llvm_unreachable("Invalid assembly expression kind!");
914}
