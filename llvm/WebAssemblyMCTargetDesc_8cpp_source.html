97// Force static initialization.
98extern "C" void LLVMInitializeWebAssemblyTargetMC() {
99 for (Target *T :
100 {&getTheWebAssemblyTarget32(), &getTheWebAssemblyTarget64()}) {
101 // Register the MC asm info.
102 RegisterMCAsmInfoFn X(*T, createMCAsmInfo);
103
104 // Register the MC instruction info.
105 TargetRegistry::RegisterMCInstrInfo(*T, createMCInstrInfo);
106
107 // Register the MC register info.
108 TargetRegistry::RegisterMCRegInfo(*T, createMCRegisterInfo);
109
110 // Register the MCInstPrinter.
111 TargetRegistry::RegisterMCInstPrinter(*T, createMCInstPrinter);
112
113 // Register the MC code emitter.
114 TargetRegistry::RegisterMCCodeEmitter(*T, createCodeEmitter);
115
116 // Register the ASM Backend.
117 TargetRegistry::RegisterMCAsmBackend(*T, createAsmBackend);
118
119 // Register the MC subtarget info.
120 TargetRegistry::RegisterMCSubtargetInfo(*T, createMCSubtargetInfo);
121
122 // Register the object target streamer.
123 TargetRegistry::RegisterObjectTargetStreamer(*T,
124 createObjectTargetStreamer);
125 // Register the asm target streamer.
126 TargetRegistry::RegisterAsmTargetStreamer(*T, createAsmTargetStreamer);
127 // Register the null target streamer.
128 TargetRegistry::RegisterNullTargetStreamer(*T, createNullTargetStreamer);
129 }
130}

132wasm::ValType WebAssembly::toValType(const MVT &Ty) {
133 switch (Ty.SimpleTy) {
134 case MVT::i32:
135 return wasm::ValType::I32;
136 case MVT::i64:
137 return wasm::ValType::I64;
138 case MVT::f32:
139 return wasm::ValType::F32;
140 case MVT::f64:
141 return wasm::ValType::F64;
142 case MVT::v16i8:
143 case MVT::v8i16:
144 case MVT::v4i32:
145 case MVT::v2i64:
146 case MVT::v4f32:
147 case MVT::v2f64:
148 return wasm::ValType::V128;
149 case MVT::exnref:
150 return wasm::ValType::EXNREF;
151 default:
152 llvm_unreachable("unexpected type");
153 }
